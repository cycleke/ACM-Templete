\subsection{Vim 配置}
\begin{lstlisting}[language=sh]
syntax enable
set syntax=on
set nobackup
set noswapfile
set noundofile
set nu
set smartindent
set cindent
set noeb
set tabstop=2
set softtabstop=2
set shiftwidth=2
set expandtab

:imap jk <Esc>

map <F5> : call Complie() <CR>
func Complie()
	exec "w"
	exec "!g++ % -o %< -g -Wall -std=gnu++14 -static"
endfunc

map <F6> : call Run() <CR>
func Run()
	exec "!./%<"
endfunc

map <F9> : call DeBug() <CR>
func DeBug()
	exec "!gdb %<"
endfunc
\end{lstlisting}
