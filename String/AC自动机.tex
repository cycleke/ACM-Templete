\begin{lstlisting}
int tr[MAX_NODE][26], fail[MAX_NODE], dep[MAX_NODE], node_c;

int add_char(int u, int id) {
  if (tr[u][id] < 0) tr[u][id] = node_c++;
  return tr[u][id];
}
void build_acam() {
  queue<int> que;
  fail[0] = 0;
  for (int i = 0; i < 26; ++i)
    if (~tr[0][i]) {
      que.push(tr[0][i]);
      fail[tr[0][i]] = 0;
    } else {
      tr[0][i] = 0;
    }
  while (!que.empty()) {
    int u = que.front(), f = fail[u];
    que.pop();
    for (int i = 0; i < 26; ++i)
      if (~tr[u][i]) {
        que.push(tr[u][i]);
        fail[tr[u][i]][i] = tr[f][i];
      } else {
        tr[u][i] = tr[f][i];
      }
  }
  for (int i = 1; i < node_c; ++i) adj[fail[i]].push_back(i);
}
\end{lstlisting}
